
Streams are implemented as sequences of StreamItem

StreamItem
{
    uuid                 : UUID
    streamuuid           : UUID
    ordinal              : Float
    generic-content : GenericContent
    filename             : String # (Re)Computed at extraction
    filepath             : String # (Re)Computed at extraction
}

StreamPrincipal
{
    "streamuuid"           : UUID,
    "description"          : String,
    "multiplicity"         : Float # multiple of 30 minutes
}
