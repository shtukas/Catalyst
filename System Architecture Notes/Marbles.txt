
Marbles

Default Attributes in data file:
    - "uuid"
    - "unixtime"
    - "domain"      : "anniversaries", "calendar", "waves", "quarks"
    - "description"
    - "type"
    - "payload"

Marble object interface
    .intialize(filepath)
    .uuid()
    .unixtime()
    .domain()
    .description()
    .type()
    .payload()
    .set(key, value)
    .getOrNull(key)
    .isStillAlive()
    .destroy()

Contents:

    "type": "Line" | "Url" | "Text" | "ClickableType" | "AionPoint"

    [type] -> [payload]

        "Line"            -> "" # empty string, the description is the payload
        "Url"             -> String
        "Text"            -> String # reference to a blob
        "ClickableType"   -> <nhash>|<dottedExtension>
        "AionPoint"       -> <nhash>

    "waves"

            - "repeatType"  : "sticky"

            - "repeatType"  : "every-n-hours"

            - "repeatType"  : "every-this-day-of-the-week"

            - "repeatType"  : "every-this-day-of-the-month"
