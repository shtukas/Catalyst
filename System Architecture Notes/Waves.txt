Wave are the todo repeating items of Catalyst. They carry a description and a schedule.

sqlite> create table _waves_ (_uuid_ text, _nereiduuid_ text, _schedule_ text, _lastDoneDateTime_ text);

Waves
{
    "uuid"             : String
    "nereiduuid"       : String
    "schedule"         : Schedule
    "lastDoneDateTime" : Datetime | null
}

Schedule
{
    "uuid"         => UUID,
    "@"            => "sticky"
    "from-hour"    => fromHour
}
{
    "uuid"         => UUID,
    "@"            => type
                        # every-n-hours
                        # every-n-days
                        # every-this-day-of-the-week
                        # every-this-day-of-the-month
    "repeat-value" => value
}
