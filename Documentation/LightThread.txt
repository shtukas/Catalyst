LightThread { 
    :uuid        : String
    :unixtime    : Integer
    :description : String
    :dailyTimeCommitment : nil or Float # value in hours
    :done        : Array[LightThreadTimeRecordItem]
    :streamuuid  : String UUID
    :targetFolderpath :
        nil                # When un-initialized
        false              # When no folder
        String, Folderpath
    :activationWeekDays : nil | Array[Weekday] # monday, tuesday, wednesday, thursday, friday, saturday, sunday
}

LightThreadTimeRecordItem: {
	"uuid"     => not particularly relevant, random
	"unixtime" => Unixtime
	"timespan" => Integer, in seconds
}
