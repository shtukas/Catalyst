
Standard interface for a Catalyst Agent

agentuid(): String
    Returns the unique id of the agent. Should never change as it is hardcoded in the stored objects

getObjects(): Array[CatalystObject]

getAllObjects(): Array[CatalystObject] # All known objects, used for search

processObjectAndCommand(object, command, isLocalCommand = true): ProcessingSignal
    [nil]                  # Nothing happens
    ["remove", objectuuid] # Object to be removed from the central repository
    ["update", object]     # Passed object to replace the existing one in repository
