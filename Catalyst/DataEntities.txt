A Data Entity is any if the four type of elements
     1. A10495 ( "catalyst-type:10014e93" )
     2. Clique ( "catalyst-type:clique" )
     3. Timeline ( "catalyst-type:timeline" )

DataEntityType = "catalyst-type:10014e93" | "catalyst-type:clique" | "catalyst-type:timeline"

All Data Entities carry
    "catalystType"      : String
    "creationTimestamp" : Float # Unixtime with decimals
    "uuid"              : String
