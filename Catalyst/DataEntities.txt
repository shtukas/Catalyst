A Data Entity is any if the four type of elements
     1. CatalystStandardTarget ( "catalyst-type:catalyst-standard-target" )
     2. Clique ( "catalyst-type:clique" )
     3. Starlight Node ( "catalyst-type:starlight-node" )

DataEntityType = "catalyst-type:catalyst-standard-target" | "catalyst-type:clique" | "catalyst-type:starlight-node"

All Data Entities carry
    "catalystType"      : String
    "creationTimestamp" : Float # Unixtime with decimals
    "uuid"              : String
